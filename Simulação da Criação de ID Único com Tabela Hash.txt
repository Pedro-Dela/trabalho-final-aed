using namespace std;

// Função para criar ID único usando tabela Hash (SHA-256)
string create_id(string data, string destination) {
    stringstream ss;
    ss << data << "-" << destination;

    // Simulação de função de hash (SHA-256)
    // Aqui estou usando uma simples formatação para demonstração
    return ss.str();
}

int main() {
    string data = "Dados importantes";
    string destination = "Dispositivo final";

    // Criando o ID único usando tabela Hash
    string id = create_id(data, destination);

    // Exibindo o ID gerado
    cout << "ID gerado para enviar dados para '" << destination << "': " << id << endl;

    return 0;
}
